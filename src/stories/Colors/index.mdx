### 2. SCSS Variables

You can use the exported `_colors.scss` file from `paytm_common_ui` with the below approach. Please note that these variables are just
static values and you will need to manage theming support on your own if you use this.

#### NPM

You can import the `_colors.scss` file from `paytm_common_ui` to any of your `scss` file like below

```css
@import '~@paytm-h5-common/paytm_common_ui/styles/colors';
```

And then you can use all the variables defined in `_colors.scss` in your app

#### MF

`_colors.scss` will be hosted in the below path

##### staging

```js
https://webappsstatic.paytmdgt.io/paytmcommonui/v1/static/styles/_colors.scss
```

##### production

```js
https://webappsstatic.paytm.com/paytmcommonui/v1/static/styles/_colors.scss
```

You can download it before your build phase and can import to your `scss` files. Below is a helper script
for downloading the style files.

```js
const https = require('https');
const fs = require('fs');
const { join } = require('path');

const files = ['_colors.scss', '_typography.scss'];
const outputLocation = join(__dirname, '..', 'src', 'commonStyles'); // edit the output path as you wish
const URL = `https://webappsstatic.paytmdgt.io/paytmcommonui/v1/static/styles/`; // modify this as per the domain you want

const printError = (file, error) => {
  console.error(
    '\x1b[31m%s: %s\x1b[0m',
    `Failed to download style file ${file}`,
    error,
  );
};

const downloadFile = file => {
  const req = https.get(`${URL}/${file}`, function (response) {
    if (response.statusCode !== 200) {
      printError(file, response.statusMessage);
      return;
    }
    const outputFile = fs.createWriteStream(`${outputLocation}/${file}`);
    response.pipe(outputFile);
    outputFile.on('finish', () => {
      outputFile.close();
      console.log('\x1b[32m%s\x1b[0m', `Downloaded ${file}`);
    });
    outputFile.on('error', error => {
      console.error(
        '\x1b[41m%s: %s\x1b[0m',
        `Failed to download style file ${file}`,
        error,
      );
    });
  });
  req.on('error', error => {
    console.error(
      '\x1b[41m%s: %s\x1b[0m',
      `Failed to download style file ${file}`,
      error,
    );
  });
};

const process = () => {
  for (const file of files) {
    downloadFile(file);
  }
};

process();
```
